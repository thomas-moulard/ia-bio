\section{Introduction}
\subsection{Limites des modèles microscopiques}

\begin{frame}
  \frametitle{Objectifs}

  \begin{block}{Description du problème}
    \begin{itemize}
    \item Etude des interactions entre protéines.
    \item Description des interactions sous forme de règle.
      Exemple: $A(b) + B(a, c) \longleftrightarrow A (b!1) . B (a!1, c) k\_on, k\_off$
    \item Problème: explosion combinatoire du nombre d'espèces chimiques/de réactions.
    \item Solution: abstraction permettant de réduire largement la taille du modèle/de la simulation.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Pourquoi une explosion combinatoire?}

  \begin{block}{Description du modèle utilisé}
    \begin{itemize}
    \item Protéines: \emph{sites} multiples aillant chacun plusieurs \emph{états} différents.
    \item Exemple: protéine phosphorylée/non-phosphorylée (état), libre/liée (site).
    \item Modèle microscopique: décrit l'ensemble des états et des réactions possibles.
    \item Impossible de travailler avec un modèle microscopique ($10^{23}$ espèces!).
    \end{itemize}
  \end{block}
\end{frame}

\subsection{Modèles orientés domaines}

\begin{frame}
  \frametitle{Vers un modèle macroscopique\ldots}

  \begin{block}{Règle de simplification}
    \begin{itemize}
    \item But: ne maintenir qu'un sous-ensemble de combinaisons domaines/sites.
    \item Possible sans perte de généralité car certains sites/domaines sont
      indépendants ou sont contrôlés dans un ordre précis.
    \item $q_i$ dépends de $q_j$ alors $q_j$ est un site controllant pour $q_i$.
    \item Pour chaque protéine complexe (géniteur), on associe plusieurs
      protéines simples (auxiliaires).
    \item Concentration d'une protéine auxiliaire Q (sites: $q_1 \ldots q_n$,
      états: $s_1 \ldots s_k$): somme des concentrations de toutes combinaisons
      de la protéine initiale aillant les mêmes états que la protéine de départ.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Limitations}

  \begin{block}{Nom de sites identiques}
    \begin{itemize}
    \item Sites considérés comme dépendants.
    \end{itemize}
  \end{block}

  \begin{block}{Dépendances implicites}
    \begin{itemize}
    \item Non considérées dans le calcul des ensembles de contrôle.
    \end{itemize}
  \end{block}

  \begin{block}{Liaison entre un site unique/partagé dans une protéine auxiliaire}
    \begin{itemize}
    \item Masse incorrecte!
    \end{itemize}
  \end{block}
\end{frame}
